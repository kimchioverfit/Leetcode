# 5. Longest Palindromic Substring


Given a string s, return the longest palindromic substring in s.



> **Example 1**
> 
> Input: s = "babad"
> 
> Output: "bab"
> 
> Explanation: "aba" is also a valid answer.


> **Example 2**
> 
> Input: s = "cbbd"
> 
> Output: "bb"

```cpp
class Solution {
public:
    string longestPalindrome(string s) {
        if (s.empty()) return "";
        
        int start = 0, end = 0;
        
        // мӨ‘к°„??кё°м??јлЎң ?‘мӘҪ??л№јк?л©ҙм„ң ?°лҰ°?ңлЎ¬??м°ҫмқҢ

            int len1 = expandAroundCenter(s, i, i); // ?Җ??кёёмқҙ ?°лҰ°?ңлЎ¬
            int len2 = expandAroundCenter(s, i, i + 1); // м§қмҲҳ кёёмқҙ ?°лҰ°?ңлЎ¬
            int len = max(len1, len2);
            
            if (len > end - start) {
                start = i - (len - 1) / 2;
                end = i + len / 2;
            }
        }
        
        return s.substr(start, end - start + 1);
    }
    
private:
    int expandAroundCenter(const string& s, int left, int right) {
        while (left >= 0 && right < s.size() && s[left] == s[right]) {
            left--;
            right++;
        }
        return right - left - 1;
    }
};
```
